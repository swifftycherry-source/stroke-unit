เนื้อสมองขาดเลือด เสี่ยงต่อสมองบวมและความพิการถาวร

สูง
ข้อมูลสนับสนุน
  • ปวดศีรษะ มึนงง อ่อนแรงซีกใดซีกหนึ่ง
  • พูดไม่ชัด กลืนลำบาก
  • NIHSS ≥ 6 คะแนน
  • CT brain: ischemic lesion
  • BP สูง SpO₂ ต่ำ
เป้าหมาย

ระยะสั้น 24–48 ชม.: ไม่มีสมองบวมเพิ่ม ระดับความรู้สึกตัวคงที่

ระยะยาว: NIHSS ลดลง ≥ 4 คะแนนใน 7 วัน ไม่เกิด secondary injury

กิจกรรมการพยาบาล
  • ประเมิน neuro signs ทุก 1 ชม. ใน 24 ชม. แรก (GCS, pupil, motor)
  • จัดท่านอน HOB 30° เพื่อลด ICP
  • ควบคุม BP: SBP <180 mmHg (ก่อน rt-PA) / <140 mmHg (หลัง rt-PA)
  • ดูแลออกซิเจน SpO₂ >94% หลีกเลี่ยง hyperthermia
  • Monitor blood glucose เป้า 140–180 mg/dL
  • ประเมิน swallowing ก่อนให้อาหาร/น้ำทุกราย

💡 Google Apps Script — ส่งข้อมูลอัตโนมัติ
// Google Apps Script สำหรับ Stroke Unit รพ.นครปฐม
// วางใน Google Apps Script แล้ว Deploy เป็น Web App

const SHEET_ID = '14bu2HUB5G3Esfmv5A43t5l5ukqE1GWIOjQwWTXWQIYE';
const SHEET_NAME = 'StrokeData';

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);

  // สร้าง Header ถ้ายังไม่มี
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      'วันที่บันทึก','ปีงบ','เดือน','รวม','Ischemic','Hemorrhagic',
      'SFT','SFT In-hos','rt-PA','rt-PA In-hos',
      'เสียชีวิต Isch','เสียชีวิต Hemo','rt-PA เสียชีวิต',
      'Refer In Isch','Refer In Hemo','Refer Out Isch','Refer Out Hemo',
      'DNT เฉลี่ย','DNT <60','DNT <45','DNT In-hos',
      'MT ติดต่อ','MT ได้ไป','MT ได้ทำ'
    ]);
  }

  sheet.appendRow([
    new Date().toLocaleDateString('th-TH'),
    data.fy, data.month, data.total,
    data.ischemic, data.hemorrhagic,
    data.sft, data.sftInh, data.rtpa, data.rtpaInh,
    data.dIsch, data.dHemo, data.dRtpa,
    data.riIsch, data.riHemo, data.roIsch, data.roHemo,
    data.dnt, data.dnt60, data.dnt45, data.dntInh,
    data.mtC, data.mtG, data.mtD
  ]);

  return ContentService.createTextOutput(
    JSON.stringify({status:'ok'})
  ).setMimeType(ContentService.MimeType.JSON);
}
